                    const opt = examOptions.find(o => o.id === examType);
                    const progress = (opt.used / opt.total) * 100;
                    return (
                      <div className="rounded-2xl p-4 mb-4 border" style={{ backgroundColor: opt.bg, borderColor: '#E5EAE8' }}>
                        <div className="flex items-center justify-between gap-3 flex-wrap mb-3">
                          <div className="flex items-center gap-2">
                            <div className="w-8 h-8 rounded-xl flex items-center justify-center" style={{ backgroundColor: opt.color }}>
                              <opt.icon size={14} className="text-white" />
                            </div>
                            <div>
                              <div className="flex items-center gap-1.5 flex-wrap">
                                <span className="text-sm" style={{ color: '#2F4F4F', fontWeight: 700 }}>{opt.name}</span>
                                {opt.betaNote && (
                                  <span className="inline-flex items-center gap-1 px-1.5 py-0.5 rounded-full text-[10px] bg-white" style={{ color: '#E67E22', fontWeight: 700 }}>
                                    <AlertCircle size={9} />
                                    베타 1회
                                  </span>
                                )}
                              </div>
                              <div className="text-[11px]" style={{ color: '#2F4F4F', opacity: 0.6, fontWeight: 500 }}>
                                {opt.desc}
                              </div>
                            </div>
                          </div>
                          <div className="flex items-center gap-3 text-[11px]" style={{ color: '#2F4F4F', opacity: 0.7, fontWeight: 500 }}>
                            <span>
                              <span className="num-tabular" style={{ color: opt.color }}>{opt.used}</span>
                              <span style={{ opacity: 0.5 }}> / </span>
                              <span className="num-tabular">{opt.total}</span>
                              <span className="ml-1">회 사용</span>
                            </span>
                          </div>
                        </div>
                        <div className="h-1.5 rounded-full overflow-hidden" style={{ backgroundColor: '#FFFFFF80' }}>
                          <div className="h-full rounded-full transition-all duration-500" style={{ width: `${progress}%`, backgroundColor: opt.color }} />
                        </div>
                      </div>
                    );
                  })()}

                  {/* 문제 카드 (모드 컬러 동기화) */}
                  <QuestionView
                    question={question}
                    selectedAnswer={selectedAnswer}
                    setSelectedAnswer={setSelectedAnswer}
                    graded={graded}
                    setGraded={setGraded}
                    isCorrect={isCorrect}
                    handleGrade={handleGrade}
                    handleReset={handleReset}
                    bannerHeader={
                      <>
                        <span className="px-3 py-1 rounded-full text-xs bg-white num-tabular" style={{ color: '#2F4F4F', fontWeight: 700 }}>
                          문항 1 / 40
                        </span>
                        <span className="text-xs text-white/90" style={{ fontWeight: 500 }}>
                          [<span style={{ color: '#FBC07A', fontWeight: 700 }}>{examOptions.find(o => o.id === examType).name}</span>] {question.topic}
                        </span>
                      </>
                    }
                    ctaColor={examOptions.find(o => o.id === examType).color}
                    ctaLabel="제출하고 다음 문항"
                    resetLabel="다음 문항 →"
                  />
                </div>
              )}
            </div>
          </main>
        </div>
      </div>
    </>
  );
}

// 기출 탭과 모의고사 탭에서 공통으로 사용하는 문제 카드 + 해설 카드
function QuestionView({
  question, selectedAnswer, setSelectedAnswer, graded, isCorrect,
  handleGrade, handleReset, bannerHeader, ctaColor, ctaLabel, resetLabel
}) {
  return (
    <div>
      {/* Question Card */}
      <div className="bg-white rounded-3xl border border-gray-100 overflow-hidden mb-4">
        {/* 다크 그라디언트 배너 */}
        <div className="px-6 py-4 flex items-center gap-3 flex-wrap" style={{ background: 'linear-gradient(135deg, #2F4F4F 0%, #3D6363 100%)' }}>
          {bannerHeader}
        </div>

        {/* 문제 본문 */}
        <div className="px-6 md:px-8 py-6">
          <p className="text-sm md:text-[15px] leading-[1.85] mb-5" style={{ color: '#2F4F4F', fontWeight: 500 }}>
            {question.body}
          </p>

          {/* 이미지 영역 */}
          <div className="mb-6 flex justify-center">
            <div className="w-full max-w-xs aspect-[3/4] rounded-2xl border-2 border-dashed flex flex-col items-center justify-center gap-2 p-6" style={{ borderColor: '#D3D9D4', backgroundColor: '#FAFBFB' }}>
              <FileText size={28} style={{ color: '#8CBDB3', opacity: 0.6 }} />
              <span className="text-xs text-center" style={{ color: '#2F4F4F', opacity: 0.55, fontWeight: 500 }}>
                흉부 X-선 사진<br />
                <span className="text-[10px]">2021년 의학과1 호흡기</span>
              </span>
            </div>
          </div>

          {/* 5지선다 */}
          <div className="space-y-2">
            {question.choices.map((c) => {
              const isSelected = selectedAnswer === c.id;
              const isAnswer = c.id === question.answer;
              let bg = '#FFFFFF', border = '#E5EAE8', textColor = '#2F4F4F', numBg = '#F5F9F8', numColor = '#2F4F4F';
              if (graded) {
                if (isAnswer) {
                  bg = '#F0F9F4'; border = '#76C7C0'; numBg = '#76C7C0'; numColor = '#FFFFFF';
                } else if (isSelected) {
                  bg = '#FDF0EE'; border = '#E67E22'; numBg = '#E67E22'; numColor = '#FFFFFF';
                }
              } else if (isSelected) {
                bg = '#F5F9F8'; border = '#8CBDB3'; numBg = '#2F4F4F'; numColor = '#FFFFFF';
              }
              return (
                <button
                  key={c.id}
                  onClick={() => !graded && setSelectedAnswer(c.id)}
                  disabled={graded}
                  className="w-full flex items-center gap-3 p-3.5 rounded-2xl text-left transition"
                  style={{
                    backgroundColor: bg,
                    border: `1.5px solid ${border}`,
                    cursor: graded ? 'default' : 'pointer',
                  }}
                >
                  <div className="flex-shrink-0 w-7 h-7 rounded-lg flex items-center justify-center text-xs num-tabular transition" style={{ backgroundColor: numBg, color: numColor }}>
                    {c.id}
                  </div>
                  <span className="flex-1 text-sm" style={{ color: textColor, fontWeight: 500 }}>
                    {c.text}
                  </span>
                  {graded && isAnswer && <Check size={16} style={{ color: '#76C7C0' }} strokeWidth={3} />}
                  {graded && isSelected && !isAnswer && <XIcon size={16} style={{ color: '#E67E22' }} strokeWidth={3} />}
                </button>
              );
            })}
          </div>
        </div>

        {/* 액션 바 */}
        <div className="px-6 md:px-8 py-4 border-t flex items-center justify-between gap-3" style={{ borderColor: '#F0F0F0', backgroundColor: '#FAFBFB' }}>
          {!graded ? (
            <>
              <span className="text-xs" style={{ color: '#2F4F4F', opacity: 0.6, fontWeight: 500 }}>
                {selectedAnswer ? `${selectedAnswer}번 선택됨` : '답을 선택해주세요'}
              </span>
              <button
                onClick={handleGrade}
                disabled={!selectedAnswer}
                className="px-6 py-2.5 rounded-full text-sm transition flex items-center gap-1.5"
                style={{
                  backgroundColor: selectedAnswer ? ctaColor : '#D3D9D4',
                  color: '#FFFFFF',
                  fontWeight: 700,
                  cursor: selectedAnswer ? 'pointer' : 'not-allowed',
                }}
              >
                {ctaLabel}
                <ArrowRight size={14} />
              </button>
            </>
          ) : (
            <>
              <div className="flex items-center gap-2">
                <div className="w-7 h-7 rounded-full flex items-center justify-center" style={{ backgroundColor: isCorrect ? '#76C7C0' : '#E67E22' }}>
                  {isCorrect ? <Check size={14} className="text-white" strokeWidth={3} /> : <XIcon size={14} className="text-white" strokeWidth={3} />}
                </div>
                <span className="text-sm" style={{ color: isCorrect ? '#5BA89F' : '#E67E22', fontWeight: 700 }}>
                  {isCorrect ? '정답입니다!' : `오답 — 정답은 ${question.answer}번`}
                </span>
              </div>
              <button
                onClick={handleReset}
                className="px-5 py-2 rounded-full text-xs border transition hover:bg-gray-50"
                style={{ color: '#2F4F4F', borderColor: '#D3D9D4', fontWeight: 700 }}
              >
                {resetLabel}
              </button>
            </>
          )}
        </div>
      </div>

      {/* 해설 카드 (채점 후 슬라이드 다운) */}
      {graded && (
        <div className="accordion-content bg-white rounded-3xl border-2 overflow-hidden" style={{ borderColor: '#76C7C0' }}>
          <div className="px-6 md:px-8 py-5 flex items-center gap-2" style={{ backgroundColor: '#F0FAF8' }}>
            <div className="w-8 h-8 rounded-xl flex items-center justify-center" style={{ backgroundColor: '#76C7C0' }}>
              <Lightbulb size={16} className="text-white" />
            </div>
            <div>
              <div className="text-sm" style={{ color: '#2F4F4F', fontWeight: 700 }}>해설</div>
              <div className="text-[11px]" style={{ color: '#2F4F4F', opacity: 0.6, fontWeight: 500 }}>
                AI가 정리한 핵심 풀이
              </div>
            </div>
          </div>
          <div className="px-6 md:px-8 py-6">
            <div className="rounded-2xl p-4 mb-4" style={{ backgroundColor: '#F5F9F8' }}>
              <div className="flex items-start gap-2">
                <Sparkles size={14} className="flex-shrink-0 mt-0.5" style={{ color: '#8CBDB3' }} />
                <p className="text-sm leading-relaxed" style={{ color: '#2F4F4F', fontWeight: 500 }}>
                  {question.explanation.summary}
                </p>
              </div>
            </div>

            <div className="text-xs mb-3 inline-flex items-center gap-1.5" style={{ color: '#2F4F4F', fontWeight: 700 }}>
              <div className="w-1 h-3 rounded-full" style={{ backgroundColor: '#E67E22' }} />
              핵심 포인트
            </div>
            <ul className="space-y-2.5">
              {question.explanation.points.map((p, i) => (
                <li key={i} className="flex items-start gap-2.5 text-sm" style={{ color: '#2F4F4F', fontWeight: 500 }}>
                  <span className="flex-shrink-0 w-5 h-5 rounded-md flex items-center justify-center text-[10px] mt-0.5 num-tabular" style={{ backgroundColor: '#FEF5EC', color: '#E67E22' }}>
                    {i + 1}
                  </span>
                  <span className="leading-relaxed">{p}</span>
                </li>
              ))}
            </ul>
          </div>
        </div>
      )}
    </div>
  );
}
